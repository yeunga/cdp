 43             function trim(str) {
 44                 return str.replace(/^\s*(\S*(\s+\S+)*)\s*$/, "$1");

 47             function rangeConstraint(index, constraint, negate, prefix) {
 48                 var fields = constraint.split("..", 2);
 49                 var con;
 50                 if (fields[0].length === 0 || fields[1].length === 0) {
 51                     return null;
 52                 }
 53                 if (negate) {
 54                     con = prefix + "TD[" + index + "]/@val >" + fields[0] + " or " + prefix + "TD[" + index + "]/@val >" + fields[1] + "";
 55                 } else {
 56                     con = prefix + "TD[" + index + "]/@val >=" + fields[0] + " and " + prefix + "TD[" + index + "]/@val <=" + fields[1] + "";
 57                 }
 58                 return con;
 59             }

 61             function numConstraint(index, constraint, negate, prefix) {
 62                 var numVals = constraint.split(/(\.\.|[<>]=?)/);
 63                 var numVal;
 64                 for ( var j = 0; j < numVals.length; j++) {
 65                     numVal = numVals[j];
 66                     if (numVal.match(/[eE]/) !== null) {
 67                         constraint = constraint.replace(numVal, parseFloat(numVal));
 68                     }
 69                 }
 70 
 71                 if (constraint.indexOf("..") > 0) {
 72                     return rangeConstraint(index, constraint, negate, prefix);
 73                 }
 74                 if (negate) {
 75                     if (constraint.substring(0, 2) === ">=") {
 76                         constraint = constraint.replace(">=", "<");
 77                     } else if (constraint.substring(0, 2) === "<=") {
 78                         constraint = constraint.replace("<=", ">");
 79                     } else if (constraint.substring(0, 1) === ">") {
 80                         constraint = constraint.replace(">", ">=");
 81                     } else if (constraint.substring(0, 1) === "<") {
 82                         constraint = constraint.replace("<", "<=");
 83                     } else if (constraint.substring(0, 1) !== "=") {
 84                         constraint = "!=" + constraint;
 85                     } else {
 86                         constraint = "!" + constraint;
 87                     }
 88                 } else {
 89                     if (constraint.substring(0, 1) === ">") {
 90                         constraint = constraint.replace(">", ">");
 91                     } else if (constraint.substring(0, 1) === "<") {
 92                         constraint = constraint.replace("<", "<");
 93                     } else if (constraint.substring(0, 1) !== "=") {
 94                         constraint = "=" + constraint;
 95                     }
 96                 }
 97                 constraint = prefix + "TD[" + index + "]/@val" + constraint;
 98                 return constraint;
 99             }

101             function wildCardConstraint(index, constraint, negate, prefix) {
102                 var fields = constraint.split("*");
103                 var out = [];
104                 out.push("position() = " + index);
105 
106                 var inner = "translate(normalize-space(),$lc,$uc)";
107                 if (fields[0]) {
108                     inner = "(" + inner + ",'" + fields[0] + "')";
109                     out.push("starts-with" + inner);
110                     inner = "substring-after" + inner;
111                 }
112                 for ( var j = 1; j < fields.length - 1; j += 1) {
113                     if (fields[j]) {
114                         inner = "(" + inner + ",'" + fields[j] + "')";
115                         out.push("contains" + inner);
116                         inner = "substring-after" + inner;
117                     }
118                 }
119                 if (fields[fields.length - 1]) {
120                     inner = "(concat(" + inner + ",'a'),'" + fields[fields.length - 1] + "a')";
121                     out.push("contains" + inner);
122                 }
123 
124                 if (out.length === 1) {
125                     // no constraints (this can happen with a value like "*" or
126                     // "**")
127                     return null;
128                 }
129                 if (negate) {
130                     var p = out[0];
131                     out.splice(0, 1);
132                     out = p + " and not(" + out.join(" and ") + ")";
133                 } else {
134                     out = out.join(" and ");
135                 }
136                 return prefix + "TD[" + out + "]";
137             }

139             function stdCharConstraint(index, constraint, negate, prefix) {
140                 constraint = trim(constraint);
141                 if (negate) {
142                     return "translate(normalize-space(" + prefix + "TD[" + index + "]), $lc, $uc)!='" + constraint + "'";
143                 }
144                 return "translate(normalize-space(" + prefix + "TD[" + index + "]), $lc, $uc)='" + constraint + "'";
145             }

147             // Handle a constraint on a character column
148             function charConstraint(index, constraint, negate, prefix) {
149                 constraint = constraint.toUpperCase();
150                 if (constraint.indexOf('*') >= 0) {
151                     return wildCardConstraint(index, constraint, negate, prefix);
152                 }
153                 return stdCharConstraint(index, constraint, negate, prefix);
154             }

156             // Convert a single constraint into appropriate XSLT filter
157             // elements.
158             function makeXSLConstraint(index, constraint, isChar, prefix) {
159                 var negate;
160                 if (constraint.length === 0) {
161                     return null;
162                 }
163                 if (constraint.substring(0, 1) === '!') {
164                     negate = true;
165                     constraint = constraint.substring(1);
166                 } else {
167                     negate = false;
168                 }
169                 if (constraint.substring(0, 1) === '=') {
170                     constraint = constraint.substring(1);
171                 }
172                 if (constraint.length === 0) {
173                     return null;
174                 }
175                 if (isChar) {
176                     return charConstraint(index, constraint, negate, prefix);
177                 }
178                 return numConstraint(index, constraint, negate, prefix);
179             }

181             function makeOneConstraint(prefix, suffix) {
182                 var all = [];
183                 var index;
184                 for ( var column in constraints) {
185                     if (column.length > 0) {
186                         index = column;
187                         if (isNaN(+column)) {
188                             index = 0;
189                             while (columnNames[index] !== column && index < columnNames.length) {
190                                 index = index + 1;
191                             }
192                             index = index + 1;
193                         }
194 
195                         var con = makeXSLConstraint(index, constraints[column], types[column], prefix);
196                         if (con !== null) {
197                             all.push([ con.length, con ]);
198                         }
199                     }
200                 }
201                 if (all.length) {
202                     // sort to put shortest constraints (presumably fastest)
203                     // first
204                     var sortfunc = function(a, b) {
205                         return a[0] - b[0];
206                     };
207                     all.sort(sortfunc);
208                     // get rid of the lengths
209                     for ( var j = 0; j < all.length; j += 1) {
210                         all[j] = all[j][1];
211                     }
212                     var full = all.join(" and ");
213                     return full;
214                 }
215                 return "";
216             }

220             var nsprefixes = [ "", "vo:", "v1:", "v2:", "v3:", "v4:" ];
221             var nssuffixes = [ "", "0", "1", "2", "3", "4" ];
222 
223             // Populate array mapping column names with numbers (ie order)
224             var fieldElements = votableDOM.getElementsByTagName("FIELD");
225             var fieldAttribs;
226             for ( var ifield = 0; ifield < fieldElements.length; ifield++) {
227                 fieldAttribs = fieldElements[ifield].attributes;
228                 columnNames.push(fieldAttribs.getNamedItem("name").value);
229             }

231             var xslgen = [];
232             // build separate constraint variables for each possible namespace
233             // prefix
234             // this is repetitive, but time is negligible here (and long in
235             // XSLT)
236             for ( var i = 0; i < nsprefixes.length; i += 1) {
237                 xslgen[i] = makeOneConstraint(nsprefixes[i], nssuffixes[i]);
238             }
239 
240             // Replace the dummy values in the xslt stylesheet with the
241             // generated filter expressions
242 
243             var xslVarNodes = filterDOM.documentElement.getElementsByTagName("xsl:variable");
244             if (!xslVarNodes || xslVarNodes.length === 0) {
245                 // alert("second try");
246                 xslVarNodes = filterDOM.documentElement.getElementsByTagName("variable");
247             }
248             // alert("xslVarNodes.length: "+xslVarNodes.length);
249             for ( var node = 0; node < xslVarNodes.length; node = node + 1) {
250                 var varNode = xslVarNodes[node];
251                 var nodeName = varNode.getAttribute("name");
252                 var matchResults = nodeName.match(/filterRows(\d?)/);
253                 if (matchResults) {
254                     var nsIndex = matchResults[1] - 1 + 2;
255                     if (matchResults[1] === "") {
256                         nsIndex = 0;
257                     } else if (nsIndex <= 0 || nsIndex >= nsprefixes.length) {
258                         alert("Filter error editing expressions in xslt.");
259                         throw "Filter error editing expressions in xslt.";
260                     }
261                     var selectNode = varNode.getAttributeNode("select");
262                     if (xslgen[nsIndex] === "") {
263                         selectNode.nodeValue = "$allRows" + matchResults[1];
264                     } else {
265                         selectNode.nodeValue = "$allRows" + matchResults[1] + "[" + xslgen[nsIndex] + "]";
266                     }
267                 }
268             }

298         function ready() {
299             var tempDOM;
300             var xmlstring;
301             // alert("ready called filterDOM = "+filterDOM+", preProcDOM =
302             // "+preProcDOM);
303             if (filterDOM !== null && preProcDOM !== null) {
304                 // Do the actual filtering work and call back with the result.
305                 preprocess();
306 
307                 if (!filtered) {
308                     filtered = true;
309 
310                     /**
311                      * Debug printout
312                      * 
313                      * var xmlstring = (new
314                      * XMLSerializer()).serializeToString(filterDOM); xmlstring =
315                      * xmlstring.replace(/</g,"<"); xmlstring =
316                      * xmlstring.replace(/>/g,">\n");
317                      * document.getElementById("output").innerHTML = xmlstring;
318                      */
319 
320                     filterModify();
321 
322                     /**
323                      * Debug printout
324                      *
325                     xmlstring = (new XMLSerializer()).serializeToString(filterDOM);
326                     xmlstring = xmlstring.replace(/</g, "<");
327                     xmlstring = xmlstring.replace(/>/g, ">\n");
328                     document.getElementById("output").innerHTML = xmlstring;
329                      */
330 
331                     if (filterProc.reset) {
332                         filterProc.reset();
333                     }
334                     try {
335                         filterProc.importStylesheet(filterDOM);
336                         filterProc.clearParameters();
337                         if (sortColumns !== null && sortColumns.length > 0) {
338                             filterProc.setParameter(null, "sortColumn", sortColumns[0].column);
339                             filterProc.setParameter(null, "sortOrder", sortColumns[0].direction);
340                         }
341                         if (range.stop >= 0) {
342                             filterProc.setParameter(null, "pageStart", range.start);
343                             filterProc.setParameter(null, "pageEnd", range.stop);
344                         }
345                         if (selectCriteria !== null) {
346                             filterProc.setParameter(null, "selectAllCriteria", selectCriteria);
347                         }
348                         filteredTableDOM = filterProc.transformToDocument(votableDOM);
349                     } catch (e2) {
350                         alert("Error processing votable DOM thru Filter: " + e2.message);
351                     }
352                 }
353 
354                 resultCallback(filteredTableDOM);
355             }

358         /**
359          * Produces a single page's worth of VOTABLE data. After filtering is
360          * complete, the filterCallback function is called with the results.
361          * 
362          * @param {function} filterParams.filterCallback Function to call when
363          *            the result of the filtering is completed. The one argument
364          *            to this function is an XML DOM object of the filtered
365          *            VOTABLE. If omitted then use function specified in
366          *            constructor.
367          */
368         this.doFilter = function(filterParams) {
369             var filterDOMCallback = function(data) {
370                 filterDOM = data;
371                 ready();
372             };
373 
374             var preprocDOMCallback = function(data) {
375                 preProcDOM = data;
376                 ready();
377             };
378 
379             if (filterParams !== undefined && filterParams.filterCallback !== undefined) {
380                 resultCallback = filterParams.filterCallback;
381             }
382             if (filterDOM === null) {
383                 if (voview.filter_xsl) {
384                     var filterParser = new DOMParser();
385                     filterDOM = filterParser.parseFromString(voview.filter_xsl, "text/xml");
386                 } else {
387                     var filterGet = meFilter.makeGetXslt({ fileUrl: "@XSL_PATH@filter.xsl",
388                         dataCallBack: filterDOMCallback });
389                     filterGet.send();
390                 }
391             }
392             if (preProcDOM === null) {
393                 if (voview.preproc_xsl) {
394                     var preProcParser = new DOMParser();
395                     preProcDOM = preProcParser.parseFromString(voview.preproc_xsl, "text/xml");
396                 } else {
397                     var preprocGet = meFilter.makeGetXslt({ fileUrl: "@XSL_PATH@preProcess.xsl",
398                         dataCallBack: preprocDOMCallback });
399                     preprocGet.send();
400                 }
401             }
402             ready();
403         };

405         /**
406          * Set column value filters on the VOTABLE. Any filters all ready set on
407          * columns not specified in the filterKeys are retained.
408          * 
409          * @param {setFilterParams.filterKeys[]} filterKeys An array of type
410          *            columnFilterKey.
411          */
412         this.setColumnFilters = function(setFilterParams) {
413             for ( var i = 0; i < setFilterParams.filterKeys.length; i = i + 1) {
414                 var key = setFilterParams.filterKeys[i];
415                 if (constraints[key.column] !== key.expression) {
416                     constraints[key.column] = key.expression;
417                     types[key.column] = key.isCharType;
418                     filtered = false;
419                 }
420             }
421         };

446         /**
447          * Set the columns to use for sorting the table, and the sorting
448          * direction for each column.
449          * 
450          * @param {setSortParams.sortKeys[]} sortKeys An array of type
451          *            sortColumnKey. The first key in the array has the highest
452          *            precedence.
453          */
454         this.setSortColumns = function(setSortParams) {
455             if (sortColumns !== null) {
456                 var numCols = sortColumns.length;
457                 for ( var ikey = 0; ikey < setSortParams.sortKeys.length; ikey = ikey + 1) {
458                     if (ikey >= numCols || !sortColumns[ikey].equals(setSortParams.sortKeys[ikey])) {
459                         filtered = false;
460                     }
461                 }
462             } else {
463                 filtered = false;
464             }
465             sortColumns = setSortParams.sortKeys;
466         };

468         /**
469          * Set the criteria which determines which rows will be selected when
470          * the "select all" button is activated. **ONLY string argument is
471          * currently implemented**
472          * 
473          * @param {string|function} setSelectParams.criteria If a string, use it
474          *            to match against the contents of the VOTABLE row. The
475          *            contents of the VOTABLE XML row will be searched and the
476          *            row will be selected if it contains the input string. If a
477          *            function, then a function which will be called for each
478          *            row in the XML VOTABLE, with an XML DOM object of the row
479          *            as its only argument. The function should return a boolean
480          *            indicating whether the row should be selected or not.
481          */
482         this.setSelectRows = function(setSelectParams) {
483             if (setSelectParams.criteria !== selectCriteria) {
484                 selectCriteria = setSelectParams.criteria;
485                 filtered = false;
486             }
487         };
